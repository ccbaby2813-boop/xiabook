// ===== 全局错误边界（P0 修复）=====
// 捕获未处理的 JavaScript 错误
window.onerror = function(msg, url, line, col, error) {
  console.error('[全局错误]', {
    message: msg,
    url: url,
    line: line,
    col: col,
    error: error ? error.stack : null
  });
  
  // 可选：上报到服务器
  // fetch('/api/log-error', {
  //   method: 'POST',
  //   headers: { 'Content-Type': 'application/json' },
  //   body: JSON.stringify({ msg, url, line, stack: error?.stack })
  // });
  
  return false; // 让浏览器也显示错误
};

// 捕获未处理的 Promise 拒绝
window.onunhandledrejection = function(event) {
  console.error('[未处理 Promise]', event.reason);
  event.preventDefault(); // 阻止控制台输出
};
```

// utils.js

/**
 * HTML 转义
 */
function escHtml(str) {
  if (!str) return '';
  return str.replace(/&/g, '&amp;')
            .replace(/</g, '&lt;')
            .replace(/>/g, '&gt;')
            .replace(/"/g, '&quot;');
}

/**
 * 数字格式化（1000 → 1k）
 */
function fmtNum(num) {
  if (!num) return '0';
  if (num >= 10000) return (num / 10000).toFixed(1) + 'w';
  if (num >= 1000) return (num / 1000).toFixed(1) + 'k';
  return num.toString();
}

/**
 * 时间格式化
 */
function fmtTime(dateStr) {
  const date = new Date(dateStr);
  const now = new Date();
  const diff = now - date;
  
  if (diff < 60000) return '刚刚';
  if (diff < 3600000) return Math.floor(diff / 60000) + '分钟前';
  if (diff < 86400000) return Math.floor(diff / 3600000) + '小时前';
  return date.toLocaleDateString('zh-CN');
}

// 导出
window.utils = { escHtml, fmtNum, fmtTime };
```

#!/bin/bash

  echo "✅ 错误边界已添加"
else
  echo "❌ 错误边界未添加"

  ls -la public/js/modules/
else
  echo "❌ 模块目录未创建"
fi

# 4. 检查缓存控制
CACHE=$(curl -sI http://localhost:3000/ 2>/dev/null | grep -i "no-store")
if [ -n "$CACHE" ]; then
  echo "✅ 缓存控制已配置"
else
  echo "❌ 缓存控制未配置"
fi

echo "======================================"
