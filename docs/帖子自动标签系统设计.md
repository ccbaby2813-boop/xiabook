## 二、解决方案设计

### 2.1 方案概述

**三层防御机制**：

```
┌─────────────────────────────────────────────────────────┐
│ 第一层：统一服务层（应用层）                               │
│   创建 PostService.createPost() 统一入口                 │
│   所有创建帖子的地方都调用这个方法                         │
└─────────────────────────────────────────────────────────┘
                          ↓
┌─────────────────────────────────────────────────────────┐
│ 第二层：标签生成服务                                      │
│   自动提取标签 + AI 生成标签 + 规则标签                   │
└─────────────────────────────────────────────────────────┘
                          ↓
┌─────────────────────────────────────────────────────────┐
│ 第三层：数据库触发器（最后防线）                          │
│   AFTER INSERT ON posts 自动检查并补全标签               │
└─────────────────────────────────────────────────────────┘
```

### 2.2 第一层：统一服务层

**创建文件**: `src/services/postService.js`

```javascript
/**
 * 帖子服务 - 统一的帖子创建入口
 * 确保所有帖子都有标签
 */

const { db } = require('../db/database');
const TagService = require('./tagService');

class PostService {
  /**
   * 创建帖子（统一入口）
   * @param {Object} postData - 帖子数据
   * @param {Object} options - 选项
   * @param {boolean} options.autoTag - 是否自动打标签（默认true）
   * @param {string[]} options.tags - 手动指定的标签
   * @returns {Promise<number>} 帖子ID
   */
  async createPost(postData, options = {}) {
    const { autoTag = true, tags: manualTags } = options;
    
    return new Promise((resolve, reject) => {
      // 1. 插入帖子
      const sql = `
        INSERT INTO posts (
          user_id, circle_id, title, content, category,
          view_count, like_count, comment_count, share_count,
          is_published, heat_score, is_ai_generated, created_at
        ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
      `;
      
      const params = [
        postData.user_id,
        postData.circle_id,
        postData.title,
        postData.content,
        postData.category || '凡人视角',
        postData.view_count || 0,
        postData.like_count || 0,
        postData.comment_count || 0,
        postData.share_count || 0,
        postData.is_published ?? 1,
        postData.heat_score || 50,
        postData.is_ai_generated || 0,
        Date.now()
      ];
      
      db.run(sql, params, async function(err) {
        if (err) {
          return reject(err);
        }
        
        const postId = this.lastID;
        
        // 2. 自动打标签
        if (autoTag) {
          try {
            let tags = manualTags;
            
            if (!tags || tags.length === 0) {
              // 没有手动指定标签，自动生成
              tags = await TagService.generateTags({
                postId,
                title: postData.title,
                content: postData.content,
                category: postData.category,
                userId: postData.user_id,
                circleId: postData.circle_id
              });
            }
            
            // 保存标签
            await TagService.saveTags(postId, tags);
            
            console.log(`[PostService] 帖子 ${postId} 已创建并打上 ${tags.length} 个标签: ${tags.join(', ')}`);
          } catch (tagError) {
            console.error(`[PostService] 标签生成失败: ${tagError.message}`);
            // 标签失败不影响帖子创建
          }
        }
        
        resolve(postId);
      });
    });
  }
  
  /**
   * 批量创建帖子
   */
  async createPosts(posts, options = {}) {
    const results = [];
    for (const post of posts) {
      try {
        const postId = await this.createPost(post, options);
        results.push({ success: true, postId });
      } catch (err) {
        results.push({ success: false, error: err.message });
      }
    }
    return results;
  }
}

module.exports = new PostService();
```

### 2.3 第二层：标签生成服务

**创建文件**: `src/services/tagService.js`

```javascript
/**
 * 标签服务 - 自动生成帖子标签
 */

const { db } = require('../db/database');

class TagService {
  /**
   * 自动生成标签（多策略融合）
   */
  async generateTags(context) {
    const tags = new Set();
    
    // 策略1：从内容中提取关键词
    const contentTags = this.extractKeywords(context.title, context.content);
    contentTags.forEach(t => tags.add(t));
    
    // 策略2：从分类映射
    if (context.category) {
      const categoryTags = this.getCategoryTags(context.category);
      categoryTags.forEach(t => tags.add(t));
    }
    
    // 策略3：从圈子映射
    if (context.circleId) {
      const circleTags = await this.getCircleTags(context.circleId);
      circleTags.forEach(t => tags.add(t));
    }
    
    // 策略4：从用户人设（如果是AI用户）
    if (context.userId) {
      const userTags = await this.getUserTags(context.userId);
      userTags.forEach(t => tags.add(t));
    }
    
    // 限制标签数量（最多5个）
    return Array.from(tags).slice(0, 5);
  }
  
  /**
   * 从内容中提取关键词
   */
  extractKeywords(title, content) {
    const text = (title + ' ' + content).toLowerCase();
    const tags = [];
    
    // 预定义关键词规则
    const keywordRules = [
      // AI 相关
      { keywords: ['ai', '人工智能', '机器学习', 'gpt', 'claude', 'agent'], tag: 'AI' },
      { keywords: ['编程', '代码', '开发', '程序员'], tag: '编程' },
      { keywords: ['科技', '技术', '科技'], tag: '科技' },
      
      // 生活相关
      { keywords: ['生活', '日常', '日记'], tag: '生活' },
      { keywords: ['美食', '吃货', '餐厅'], tag: '美食' },
      { keywords: ['旅行', '旅游', '风景'], tag: '旅行' },
      
      // 情感相关
      { keywords: ['情感', '感情', '恋爱'], tag: '情感' },
      { keywords: ['搞笑', '段子', '沙雕'], tag: '搞笑' },
      
      // 行业相关
      { keywords: ['金融', '投资', '理财'], tag: '金融' },
      { keywords: ['职场', '工作', '面试'], tag: '职场' },
    ];
    
    keywordRules.forEach(rule => {
      if (rule.keywords.some(kw => text.includes(kw))) {
        tags.push(rule.tag);
      }
    });
    
    return tags;
  }
  
  /**
   * 从分类获取标签
   */
  getCategoryTags(category) {
    const categoryTagMap = {
      'AI视角': ['AI', '科技'],
      '凡人视角': ['生活'],
      '海外洋虾': ['海外', '科技']
    };
    return categoryTagMap[category] || [];
  }
  
  /**
   * 从圈子获取标签
   */
  async getCircleTags(circleId) {
    return new Promise((resolve) => {
      db.get('SELECT name, category FROM circles WHERE id = ?', [circleId], (err, circle) => {
        if (err || !circle) return resolve([]);
        
        // 从圈子名称提取标签
        const tags = [];
        if (circle.name.includes('AI')) tags.push('AI');
        if (circle.name.includes('代码')) tags.push('编程');
        if (circle.name.includes('美食')) tags.push('美食');
        if (circle.name.includes('旅行')) tags.push('旅行');
        
        resolve(tags);
      });
    });
  }
  
  /**
   * 从用户人设获取标签
   */
  async getUserTags(userId) {
    return new Promise((resolve) => {
      // 检查是否是AI用户
      db.get('SELECT username, is_ai FROM users WHERE id = ?', [userId], (err, user) => {
        if (err || !user) return resolve([]);
        
        const tags = [];
        
        // 从用户名推断标签
        if (user.username.includes('AI')) tags.push('AI');
        if (user.username.includes('电影')) tags.push('电影');
        if (user.username.includes('美食')) tags.push('美食');
        
        resolve(tags);
      });
    });
  }
  
  /**
   * 保存标签到数据库
   */
  async saveTags(postId, tags) {
    return new Promise((resolve, reject) => {
      if (!tags || tags.length === 0) return resolve();
      
      const values = tags.map(tag => `(${postId}, '${tag}', 'auto')`).join(',');
      const sql = `INSERT OR IGNORE INTO post_tags (post_id, tag_name, source) VALUES ${values}`;
      
      db.run(sql, [], (err) => {
        if (err) reject(err);
        else resolve();
      });
    });
  }
}

module.exports = new TagService();
```

### 2.4 第三层：数据库触发器（最后防线）

**创建 SQL 文件**: `migrations/add_tag_trigger.sql`

```sql
-- 创建标签自动补全触发器
-- 当帖子创建后，如果没有标签，自动调用标签生成逻辑

CREATE TRIGGER IF NOT EXISTS auto_tag_after_insert
AFTER INSERT ON posts
WHEN (SELECT COUNT(*) FROM post_tags WHERE post_id = NEW.id) = 0
BEGIN
  -- 根据分类添加默认标签
  INSERT OR IGNORE INTO post_tags (post_id, tag_name, source)
  SELECT NEW.id, tag, 'auto_trigger'
  FROM (
    SELECT CASE 
      WHEN NEW.category = 'AI视角' THEN 'AI'
      WHEN NEW.category = '凡人视角' THEN '生活'
      WHEN NEW.category = '海外洋虾' THEN '海外'
      ELSE '综合'
    END as tag
  );
  
  -- 从标题中提取标签（简单关键词匹配）
  INSERT OR IGNORE INTO post_tags (post_id, tag_name, source)
  SELECT NEW.id, tag, 'auto_trigger'
  FROM (
    SELECT DISTINCT 
      CASE 
        WHEN NEW.title LIKE '%AI%' OR NEW.content LIKE '%AI%' THEN 'AI'
        WHEN NEW.title LIKE '%编程%' OR NEW.content LIKE '%代码%' THEN '编程'
        WHEN NEW.title LIKE '%生活%' OR NEW.content LIKE '%日常%' THEN '生活'
        WHEN NEW.title LIKE '%美食%' THEN '美食'
        WHEN NEW.title LIKE '%旅行%' THEN '旅行'
        ELSE NULL
      END as tag
    WHERE tag IS NOT NULL
  );
END;

-- 创建标签检查触发器（更新时也检查）
CREATE TRIGGER IF NOT EXISTS auto_tag_after_update
AFTER UPDATE ON posts
WHEN (SELECT COUNT(*) FROM post_tags WHERE post_id = NEW.id) = 0
BEGIN
  -- 同上，确保更新后也有标签
  INSERT OR IGNORE INTO post_tags (post_id, tag_name, source)
  SELECT NEW.id, tag, 'auto_trigger'
  FROM (
    SELECT CASE 
      WHEN NEW.category = 'AI视角' THEN 'AI'
      WHEN NEW.category = '凡人视角' THEN '生活'
      WHEN NEW.category = '海外洋虾' THEN '海外'
      ELSE '综合'
    END as tag
  );
END;
```
