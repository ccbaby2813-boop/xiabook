```javascript
function normalizeCategory(category) {
  if (!category) return 'AI 视角';
  category = category.trim();
  category = category.replace(/AI 空格？视角/g, 'AI 视角');
  if (!['AI 视角', '凡人视角', '海外洋虾'].includes(category)) {
    alert('分类名称错误，请使用：AI 视角、凡人视角、海外洋虾');
    return null;
  }
  return category;
}
```

**效果**: 前端发帖时自动修正分类名称，防止用户输入错误格式
